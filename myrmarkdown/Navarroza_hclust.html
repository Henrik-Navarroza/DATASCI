setwd("C:/Users/Admin/downloads")
library(cluster)
morph_data <- read.csv("morphology_data.csv")
morph_clean <- morph_data[, !(names(morph_data) %in% c("Pop", "Sex"))]
morph_clean <- na.omit(morph_clean)
dist_mat <- dist(scale(morph_clean), method = "euclidean")
k_range <- 2:10
sil_ward <- numeric(length(k_range))
sil_single <- numeric(length(k_range))
for (i in seq_along(k_range)) {
  k <- k_range[i]
  
  # Ward's Method
  h_ward <- hclust(dist_mat, method = "ward.D2")
  sil_ward[i] <- mean(silhouette(cutree(h_ward, k), dist_mat)[, 3])
  # Single Linkage
  h_single <- hclust(dist_mat, method = "single")
  sil_single[i] <- mean(silhouette(cutree(h_single, k), dist_mat)[, 3])
}

plot(k_range, sil_ward, type = "b", pch = 19, col = "blue", 
     ylim = c(min(c(sil_ward, sil_single)), max(c(sil_ward, sil_single))),
     xlab = "Number of clusters (K)", 
     ylab = "Average Silhouette Width",
     main = "Silhouette Scores: Ward vs. Single Linkage")

lines(k_range, sil_single, type = "b", pch = 19, col = "red")
legend("topright", legend = c("Ward's", "Single Linkage"), 
       col = c("blue", "red"), lty = 1, pch = 19)

hclust_ward <- hclust(dist_mat, method = "ward.D2")
hclust_single <- hclust(dist_mat, method = "single")
plot(hclust_ward, main = "Dendrogram: Ward D2", sub = "", xlab = "Lizard Index",ylab = "Height")
rect.hclust(hclust_ward, k = 3, border = c("red","blue","green"))

plot(hclust_single, main = "Dendrogram: Single Linkage", xlab = "Lizard Index", sub = "", ylab = "Height")
rect.hclust(hclust_single, k = 3, border = c("red","blue","green"))
